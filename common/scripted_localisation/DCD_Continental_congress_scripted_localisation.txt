	}

	text = {
		trigger = {
			has_country_flag = doing_contcon_military
		}
		localization_key = DCD_contcon_military_vote
	}
	text = {
		trigger = {
			has_country_flag = doing_contcon_faction
		}
		localization_key = DCD_contcon_faction_vote
	}
	text = {
		trigger = {
			NOT = {
				OR = {
					has_country_flag = doing_contcon_economic
					has_country_flag = doing_contcon_faction

					has_country_flag = doing_contcon_political
				}
			}
		}
		localization_key = DCD_no_current_vote
	}
