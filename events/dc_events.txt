country_event = {
	id = continental_congress.6
	title = continental_congress.6.t
	desc = continental_congress.6.d
	is_triggered_only = yes
	option = {
		name = continental_congress.6.a
		ai_chance = {
			base = 1
		}
		if = { #Military Collaboration Accepted
			limit = {
				DCD = {
					has_country_flag = contcon_military_idea_pending
				}
			}
			add_ideas = DC_military_cooperation_idea
		}
		if = { #Federation Accepted
			limit = {
				NOT = {
					tag = DCD
				}
				DCD = {
					has_country_flag = contcon_federation_pending
				}
			}
			DCD = {
				set_autonomy = {
					target = ROOT
					autonomy_state = autonomy_federal_state
					freedom_level = 0
				}
			}
		}
		if = { #Economic Collaboration Accepted
			limit = {
				DCD = {
					has_country_flag = contcon_economic_idea_pending
				}
			}
			add_ideas = DC_economic_cooperation_idea
		}
		if = { #Faction Accepted
			limit = {
				DCD = {
					is_faction_leader = yes
				}
			}
			DCD = {
				add_to_faction = ROOT
			}
		}
	}
	option = {
		name = continental_congress.6.b
		ai_chance = {
			base = 0
		}
		if = { #Federation Was Accepted - Back Out
			limit = {
				DCD = {
					has_country_flag = contcon_federation_pending
				}
			}
			custom_effect_tooltip = DCD_congressional_retaliation_tt
			hidden_effect = {
				DCD = {
					create_wargoal = {
						target = ROOT
						type = puppet_wargoal_focus
					}
				}
			}
		}
	}
}
